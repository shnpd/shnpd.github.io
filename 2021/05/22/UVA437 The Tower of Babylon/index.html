йўҳзӣ®дёӯжңүдёӨеҸҘиҜқжҜ”иҫғйҮҚиҰҒпјҢдёҖжҳҜжҜҸз§Қз«Ӣж–№дҪ“йғҪжңүж— з©·еӨҡдёӘпјҢдәҢжҳҜеҸҜд»ҘиҮӘиЎҢйҖүжӢ©е“ӘдёҖжқЎиҫ№дҪңдёәй«ҳпјҢжүҖд»ҘеҪ“иҫ“е…Ҙдёәnз§Қз«Ӣж–№дҪ“ж—¶пјҢеҸҜдҫӣжҲ‘д»¬йҖүжӢ©зҡ„з«Ӣж–№дҪ“дёӘж•°дёҖе…ұжҳҜ3*nдёӘпјҢжҜҸиҫ“е…ҘдёҖз§Қз«Ӣж–№дҪ“пјҢеҜ№еә”зҡ„е°ұжңүдёүз§Қз«Ӣж–№дҪ“дҫӣжҲ‘д»¬йҖүжӢ©пјҲеҲҶеҲ«д»Ҙиҫ“е…Ҙзҡ„дёүдёӘж•°дҪңдёәй«ҳпјүгҖӮ
з„¶еҗҺж №жҚ®з«Ӣж–№дҪ“д№Ӣй—ҙиғҪеҗҰж‘һеңЁдёҖиө·зҡ„е…ізі»еҸҜд»Ҙе»әз«Ӣжңүеҗ‘ж— зҺҜеӣҫпјҲDAGпјүпјҢеҲ©з”ЁйӮ»жҺҘзҹ©йҳөеӯҳеӮЁгҖӮ

#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int MAX = 1000;
int n;
struct rec
{
    int a, b, h;
    rec(int x = 0, int y = 0, int z = 0)
    {
        a = x;
        b = y;
        h = z;
    }
};
rec recs[200];
int t1, t2, t3;
int G[200][200];
bool judge(rec x, rec y)//xиғҪеҗҰж”ҫеңЁyдёҠйқў
{
    if ((x.a < y.a && x.b < y.b) || (x.a < y.b && x.b < y.a))
        return true;
    return false;
}
void graph()//еҰӮжһңiиғҪж”ҫеңЁjдёҠйқўеҲҷG[i][j]=1
{
    memset(G, 0, sizeof(G));
    for (int i = 0; i < 3 * n; i++)
    {
        for (int j = 0; j < 3 * n; j++)
        {
            if (judge(recs[i], recs[j]))
                G[i][j] = 1;
        }
    }
}
int dp[200];
int d(int i)
{
    if (dp[i] > 0)
        return dp[i];
    dp[i] = recs[i].h;
    for (int j = 0; j < 3 * n; j++)
    {
        if (G[j][i])
            dp[i] = max(dp[i], d(j) + recs[i].h);
    }
    return dp[i];
}
int main()
{
    int kase = 0;
    while (cin >> n && n)
    {
        for (int i = 0; i < n; i++)
        {
            cin >> t1 >> t2 >> t3;
            recs[3 * i].a = t1;
            recs[3 * i].b = t2;
            recs[3 * i].h = t3;

            recs[3 * i + 1].a = t2;
            recs[3 * i + 1].b = t3;
            recs[3 * i + 1].h = t1;

            recs[3 * i + 2].a = t3;
            recs[3 * i + 2].b = t1;
            recs[3 * i + 2].h = t2;
        }
        graph();
        memset(dp, 0, sizeof(dp));
        int ans = 0;
        for (int t = 0; t < 3 * n; t++)
        {
            ans = max(ans, d(t));
        }
        printf("Case %d: maximum height = %d\n", ++kase, ans);
    }
}
